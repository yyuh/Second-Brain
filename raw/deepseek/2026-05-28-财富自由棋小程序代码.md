App({
  globalData: {
    userInfo: null,
    soundEnabled: true,
    musicEnabled: true,
    vibrateEnabled: true
  },

  onLaunch() {
    console.log('财富自由棋启动')
    const userInfo = wx.getStorageSync('userInfo')
    if (userInfo) {
      this.globalData.userInfo = userInfo
    }
    const settings = wx.getStorageSync('gameSettings')
    if (settings) {
      this.globalData.soundEnabled = settings.soundEnabled !== false
      this.globalData.musicEnabled = settings.musicEnabled !== false
      this.globalData.vibrateEnabled = settings.vibrateEnabled !== false
    }
  },

  playEffect(effectName) {
    if (!this.globalData.soundEnabled) return
    console.log('音效:', effectName)
  },

  vibrate() {
    if (!this.globalData.vibrateEnabled) return
    wx.vibrateShort({ type: 'light' })
  },

  updateSettings(settings) {
    if (settings.soundEnabled !== undefined) {
      this.globalData.soundEnabled = settings.soundEnabled
    }
    if (settings.musicEnabled !== undefined) {
      this.globalData.musicEnabled = settings.musicEnabled
    }
    if (settings.vibrateEnabled !== undefined) {
      this.globalData.vibrateEnabled = settings.vibrateEnabled
    }
    wx.setStorageSync('gameSettings', {
      soundEnabled: this.globalData.soundEnabled,
      musicEnabled: this.globalData.musicEnabled,
      vibrateEnabled: this.globalData.vibrateEnabled
    })
  }
})

const { getSignInfo, doSign } = require('../../utils/sign.js')

Page({
  data: {
    userName: '玩家',
    hasSave: false,
    signInfo: null,
    hotCareers: [
      { id: 1, name: '教师', icon: '👩‍🏫', round: 1 },
      { id: 2, name: '医生', icon: '👨‍⚕️', round: 1 },
      { id: 3, name: '律师', icon: '👨‍⚖️', round: 1 }
    ]
  },

  onLoad() {
    const userInfo = wx.getStorageSync('userInfo')
    if (userInfo && userInfo.nickname) {
      this.setData({ userName: userInfo.nickname })
    }
    this.loadSignInfo()
    const saveData = wx.getStorageSync('gameSave')
    this.setData({ hasSave: !!saveData })
  },

  onShow() {
    this.loadSignInfo()
    const saveData = wx.getStorageSync('gameSave')
    this.setData({ hasSave: !!saveData })
  },

  loadSignInfo() {
    this.setData({ signInfo: getSignInfo() })
  },

  doSign() {
    const result = doSign()
    if (result.success) {
      let userCash = wx.getStorageSync('userCash') || 0
      userCash += result.reward.reward
      wx.setStorageSync('userCash', userCash)
      wx.showToast({ title: result.message, icon: 'success' })
      this.loadSignInfo()
    } else {
      wx.showToast({ title: result.message, icon: 'none' })
    }
  },

  continueGame() {
    wx.navigateTo({ url: '/pages/game/game?loadSave=true' })
  },

  startNewGame() {
    wx.navigateTo({ url: '/pages/career/career' })
  },

  selectCareer(e) {
    const career = e.currentTarget.dataset.career
    wx.setStorageSync('selectedCareer', career)
    wx.navigateTo({ url: '/pages/game/game' })
  },

  showSaveManage() {
    const saveData = wx.getStorageSync('gameSave')
    if (saveData) {
      wx.showModal({
        title: '存档管理',
        content: `检测到存档：${saveData.career?.name || '未知'}职业，第${saveData.currentRound || 1}圈\n是否删除存档？`,
        confirmText: '删除',
        cancelText: '取消',
        success: (res) => {
          if (res.confirm) {
            wx.removeStorageSync('gameSave')
            this.setData({ hasSave: false })
            wx.showToast({ title: '存档已删除', icon: 'success' })
          }
        }
      })
    } else {
      wx.showToast({ title: '暂无存档', icon: 'none' })
    }
  },

  showRules() {
    wx.showModal({
      title: '游戏规则',
      content: '🎲 游戏规则\n\n1. 掷骰子走格子，触发事件\n2. 买入资产增加被动收入\n3. 被动收入 > 总支出 = 财务自由\n4. 现金 < 0 且 月现金流 < 0 = 破产\n5. 每日新闻影响资产价格\n6. 破产后可以分享续命',
      showCancel: false,
      confirmText: '知道了'
    })
  },

  logout() {
    wx.showModal({
      title: '退出登录',
      content: '确定要退出登录吗？',
      success: (res) => {
        if (res.confirm) {
          wx.removeStorageSync('userInfo')
          wx.removeStorageSync('selectedCareer')
          wx.removeStorageSync('gameSave')
          this.setData({ userName: '玩家', hasSave: false })
          wx.showToast({ title: '已退出', icon: 'success' })
        }
      }
    })
  }
})

const { newsDatabase } = require('../../utils/news.js')

const { needTutorial, markTutorialShown, getTutorialSteps } = require('../../utils/tutorial.js')

Page({
  data: {
    career: null,
    cash: 0,
    monthCashFlow: 0,
    passiveIncome: 0,
    totalExpense: 0,
    financialFreedom: 0,
    currentRound: 1,
    currentPosition: 0,
    isRolling: false,
    showEventPanel: false,
    showNewsModal: false,
    showAssets: false,
    showRichDad: false,
    showTutorial: false,
    eventTitle: '',
    eventStory: '',
    eventButtons: [],
    currentEventData: null,
    displayGrids: [],
    todayNews: [],
    stocks: [],
    stocksWithValue: [],
    unlockedAchievements: [],
    luckyStreak: 0,
    richDadAdvice: '',
    richDadSummary: '',
    tutorialStep: 0,
    tutorialSteps: [],
    tutorialHighlight: {},
    tutorialTitle: '',
    tutorialContent: '',
    tutorialPosition: 'bottom'
  },

  onLoad(options) {
    let career = wx.getStorageSync('selectedCareer')
    if (!career) {
      wx.showToast({ title: '请先选择职业', icon: 'none' })
      setTimeout(() => wx.navigateBack(), 1500)
      return
    }
    
    const unlocked = wx.getStorageSync('unlockedAchievements') || []
    this.setData({ unlockedAchievements: unlocked })
    
    if (options.loadSave) {
      const saveData = wx.getStorageSync('gameSave')
      if (saveData) {
        career = saveData.career
        this.setData(saveData)
        this.updateRingGrids(saveData.currentPosition || 0)
        this.calculateFreedom()
        this.loadTodayNews()
        return
      }
    }
    
    this.initGame(career)
    this.loadTodayNews()
  },

  initGame(career) {
    this.setData({
      career: career,
      cash: career.cash,
      monthCashFlow: career.salary - career.expense,
      passiveIncome: 0,
      totalExpense: career.expense,
      financialFreedom: 0,
      currentRound: 1,
      currentPosition: 0,
      stocks: [],
      luckyStreak: 0
    })
    this.updateRingGrids(0)
    this.calculateFreedom()
    
    if (needTutorial()) {
      this.startTutorial()
    }
  },

  startTutorial() {
    const steps = getTutorialSteps()
    this.setData({
      showTutorial: true,
      tutorialSteps: steps,
      tutorialStep: 0
    })
    this.showTutorialStep(0)
  },

  showTutorialStep(index) {
    const step = this.data.tutorialSteps[index]
    if (!step) {
      this.endTutorial()
      return
    }
    
    const query = wx.createSelectorQuery()
    query.select(step.target).boundingClientRect()
    query.exec((res) => {
      if (res[0]) {
        this.setData({
          tutorialHighlight: res[0],
          tutorialTitle: step.title,
          tutorialContent: step.content,
          tutorialPosition: step.position
        })
      } else {
        this.nextTutorial()
      }
    })
  },

  nextTutorial() {
    const nextStep = this.data.tutorialStep + 1
    if (nextStep >= this.data.tutorialSteps.length) {
      this.endTutorial()
    } else {
      this.setData({ tutorialStep: nextStep })
      this.showTutorialStep(nextStep)
    }
  },

  endTutorial() {
    this.setData({ showTutorial: false })
    markTutorialShown()
  },

  loadTodayNews() {
    const today = new Date().toISOString().slice(0, 10)
    const todayNews = newsDatabase.filter(n => n.date === today)
    this.setData({ todayNews })
  },

  showNews() {
    this.setData({ showNewsModal: true })
  },

  closeNews() {
    this.setData({ showNewsModal: false })
  },

  updateRingGrids(position) {
    const gridTypes = [
      { type: 'bank', title: '银行', icon: '🏦' },
      { type: 'stock', title: '股票', icon: '📈' },
      { type: 'expense', title: '支出', icon: '💸' },
      { type: 'bank', title: '银行', icon: '🏦' },
      { type: 'stock', title: '股票', icon: '📈' },
      { type: 'expense', title: '支出', icon: '💸' }
    ]
    
    const displayGrids = []
    const radius = 200
    for (let i = -3; i <= 3; i++) {
      const gridId = (position + i + 60) % 60
      const angle = i * 20
      const radian = angle * Math.PI / 180
      const x = 250 + radius * Math.sin(radian)
      const y = 280 + -radius * Math.cos(radian)
      const scale = 1 - Math.abs(i) * 0.1
      
      displayGrids.push({
        id: gridId,
        type: gridTypes[gridId % gridTypes.length].type,
        icon: gridTypes[gridId % gridTypes.length].icon,
        title: gridTypes[gridId % gridTypes.length].title,
        x: x,
        y: y,
        scale: scale,
        zIndex: 10 - Math.abs(i),
        isCenter: i === 0
      })
    }
    this.setData({ displayGrids })
  },

  calculateFreedom() {
    const { passiveIncome, totalExpense } = this.data
    let freedom = 0
    if (totalExpense > 0) {
      freedom = Math.min(Math.floor((passiveIncome / totalExpense) * 100), 100)
    }
    this.setData({ financialFreedom: freedom })
    
    if (passiveIncome > totalExpense) {
      this.gameVictory()
    }
  },

  rollDice() {
    if (this.data.isRolling || this.data.showEventPanel || this.data.showNewsModal || this.data.showAssets || this.data.showRichDad) return
    
    this.setData({ isRolling: true })
    getApp().vibrate()
    
    const result = Math.floor(Math.random() * 6) + 1
    
    let luckyStreak = this.data.luckyStreak
    if (result === 6) {
      luckyStreak++
      this.setData({ luckyStreak })
    } else {
      this.setData({ luckyStreak: 0 })
    }
    
    let rollCount = 0
    const rollInterval = setInterval(() => {
      const randomPoint = Math.floor(Math.random() * 6) + 1
      this.setData({ diceResult: randomPoint })
      rollCount++
      if (rollCount >= 10) {
        clearInterval(rollInterval)
        setTimeout(() => {
          this.setData({ isRolling: false })
          this.movePlayer(result)
        }, 200)
      }
    }, 50)
  },

      this.bankSettlement()

    this.setData({ 
      currentPosition: newPos, 
      currentRound: newRound

    if (this.data.career.name === '设计师') amount += 200

    this.saveGameProgress()

    const eventData = this.data.currentEventData
    
    this.setData({ showEventPanel: false })
    
    if (type === 'buy') {
      this.buyStock(eventData.stock)

      this.updateCash(-eventData.cost, `支出 ¥${eventData.cost}`)
    }
  },

  buyStock(stock) {
    let cost = stock.price * 100
    let discountMsg = ''
    
    if (this.data.career.name === '程序员') {
      cost = Math.floor(cost * 0.9)
      discountMsg = '（程序员9折优惠）'
    }
    
    if (this.data.cash < cost) {
      wx.showToast({ title: '现金不足', icon: 'none' })
      return
    }
    
    const stocks = [...this.data.stocks]
    const existing = stocks.find(s => s.name === stock.name)
    if (existing) {
      existing.shares += 100
      existing.totalCost += cost
    } else {
      stocks.push({ 
        name: stock.name, 
        shares: 100, 
        buyPrice: stock.price, 
        totalCost: cost, 
        cashFlow: stock.dividend * 100 
      })
    }
    
    this.setData({ 
      cash: this.data.cash - cost, 
      stocks: stocks 
    })
    this.updateCashFlow(stock.dividend * 100)
    
    wx.showToast({ title: `买入成功${discountMsg}`, icon: 'success' })
    this.saveGameProgress()
    this.checkAndUnlockAchievements()
  },

    this.setData({ cash: newCash })

      this.showBankruptcyModal()
    }
    this.saveGameProgress()
    this.checkAndUnlockAchievements()
  },

      passiveIncome: newPassive, 
      monthCashFlow: newCashFlow 
    })
    this.calculateFreedom()
  },

      totalExpense: this.data.totalExpense,
      career: this.data.career,
      luckyStreak: this.data.luckyStreak
    }
    
    const newlyUnlocked = checkAchievements(gameData, this.data.unlockedAchievements)
    
    newlyUnlocked.forEach(ach => {
      const unlocked = [...this.data.unlockedAchievements, ach.id]
      this.setData({ unlockedAchievements: unlocked })
      wx.setStorageSync('unlockedAchievements', unlocked)
      this.updateCash(ach.reward, `成就解锁：${ach.name} +¥${ach.reward}`)
      showAchievementToast(ach)
    })
  },

  showBankruptcyModal() {
    wx.showModal({
      title: '💀 破产危机',

      confirmText: '分享续命',
      cancelText: '放弃',
      success: (res) => {
        if (res.confirm) {
          const reviveAmount = Math.floor(Math.random() * 15000) + 5000
          this.updateCash(reviveAmount, `分享续命 +¥${reviveAmount}`)
          wx.showToast({ title: `获得¥${reviveAmount}救命资金`, icon: 'success' })
        } else {
          this.gameOver()
        }
      }
    })
  },

  showAssetsList() {
    const stocksWithValue = this.data.stocks.map(stock => {
      const currentPrice = this.getStockCurrentPrice(stock.name)
      const currentValue = currentPrice * stock.shares
      const profit = currentValue - stock.totalCost
      return {
        ...stock,
        currentPrice: currentPrice,
        currentValue: currentValue,
        profit: profit,
        profitPercent: ((profit / stock.totalCost) * 100).toFixed(1)
      }
    })
    
    this.setData({ 
      showAssets: true,
      stocksWithValue: stocksWithValue
    })
  },

  getStockCurrentPrice(name) {
    let price = 11
    if (name === '农业股票') price = 11
    if (name === '科技股票') price = 25
    if (name === '能源股票') price = 18
    if (name === '医药股票') price = 30
    return price
  },

  sellAsset(e) {
    const { index } = e.currentTarget.dataset
    const stock = this.data.stocksWithValue[index]
    
    wx.showModal({
      title: '确认变卖',
      content: `${stock.name}\n持有：${stock.shares}股\n当前价：¥${stock.currentPrice}\n总价值：¥${stock.currentValue}\n收益：${stock.profit >= 0 ? '+' : ''}¥${stock.profit} (${stock.profitPercent}%)`,
      confirmText: '卖出',
      cancelText: '取消',
      success: (res) => {
        if (res.confirm) {
          const stocks = [...this.data.stocks]
          const removed = stocks.splice(index, 1)[0]
          
          this.updateCash(stock.currentValue, `卖出${stock.name} +¥${stock.currentValue}`)
          this.updateCashFlow(-removed.cashFlow)
          this.setData({ stocks })
          
          wx.showToast({ title: '变卖成功', icon: 'success' })
          this.showAssetsList()
        }
      }
    })
  },

  closeAssets() {
    this.setData({ showAssets: false })
  },

  askRichDad() {
    const eventData = this.data.currentEventData
    const advice = this.generateRichDadAdvice(eventData)
    
    this.setData({
      showRichDad: true,
      richDadAdvice: advice.advice,
      richDadSummary: advice.summary
    })
  },

  generateRichDadAdvice(eventData) {
    const cash = this.data.cash
    const financialFreedom = this.data.financialFreedom
    
    if (eventData.type === 'stock') {
      const stock = eventData.stock
      
      if (cash < stock.price * 100) {
        return {
          advice: `孩子，你想买${stock.name}，但你的现金只有¥${cash}，不够买一手。\n\n记住：永远不要借钱投资。先积累本金，再考虑投资。`,
          summary: '投资要用闲钱，不要借钱投资。'
        }
      } else if (financialFreedom < 30) {
        return {
          advice: `你的财务自由度只有${financialFreedom}%，还在老鼠赛跑的初期。\n\n${stock.name}今日价格¥${stock.price}，处于${stock.range}的中间位置。\n\n我的建议是：先建立紧急备用金，再考虑股票投资。`,
          summary: '先建立紧急备用金，再考虑高风险投资。'
        }
      } else if (stock.price < 15) {
        return {
          advice: `${stock.name}今日价格¥${stock.price}，处于历史低位。\n\n别人恐惧时我贪婪。现在可能是买入的好时机。`,
          summary: '低价买入是投资的好时机，但要控制仓位。'
        }

          advice: `${stock.name}今日价格¥${stock.price}，处于中高位。\n\n投资不是赌博。等待好的价格再入场。`,
          summary: '等待好价格再买入，不要追高。'
        }
      }
    } else if (eventData.type === 'expense') {
      return {
        advice: `这笔¥${eventData.cost}的支出，你想清楚了吗？\n\n富人最后才买奢侈品。他们先建立资产，用资产产生的收入来买单。`,
        summary: '控制消费欲望，把钱花在能生钱的地方。'
      }
    }
    
    return {
      advice: '投资需要耐心和智慧。多学习，多思考。',
      summary: '学习是投资的第一步。'
    }
  },

  closeRichDad() {
    this.setData({ showRichDad: false })
  },

  saveGameProgress() {
    const saveData = {
      career: this.data.career,
      cash: this.data.cash,
      monthCashFlow: this.data.monthCashFlow,
      passiveIncome: this.data.passiveIncome,
      totalExpense: this.data.totalExpense,
      currentRound: this.data.currentRound,
      currentPosition: this.data.currentPosition,
      stocks: this.data.stocks
    }
    wx.setStorageSync('gameSave', saveData)
  },

  gameVictory() {
    wx.showModal({
      title: '🎉 胜利！',
      content: '恭喜您实现财务自由！\n您已跳出老鼠赛跑！',
      showCancel: false,

        wx.removeStorageSync('gameSave')
        wx.navigateBack()
      }

      title: '💀 游戏结束',

        wx.removeStorageSync('gameSave')

const { achievements } = require('../../utils/achievements.js')

Page({
  data: {
    userInfo: {},
    stats: {
      totalGames: 0,
      winGames: 0,
      maxFreedom: 0,
      totalAssets: 0
    },
    achievements: [],
    soundEnabled: true,
    musicEnabled: true,
    vibrateEnabled: true
  },

  onLoad() {
    const app = getApp()
    this.setData({
      soundEnabled: app.globalData.soundEnabled,
      musicEnabled: app.globalData.musicEnabled,
      vibrateEnabled: app.globalData.vibrateEnabled
    })
    
    const userInfo = wx.getStorageSync('userInfo') || {}
    const stats = wx.getStorageSync('gameStats') || {
      totalGames: 0,
      winGames: 0,
      maxFreedom: 0,
      totalAssets: 0
    }
    
    const unlockedIds = wx.getStorageSync('unlockedAchievements') || []
    const achievementsWithStatus = achievements.map(ach => ({
      ...ach,
      unlocked: unlockedIds.includes(ach.id)
    }))
    
    this.setData({ userInfo, stats, achievements: achievementsWithStatus })
  },

  toggleMusic(e) {
    const value = e.detail.value
    this.setData({ musicEnabled: value })
    getApp().updateSettings({ musicEnabled: value })
  },

  toggleSound(e) {
    const value = e.detail.value
    this.setData({ soundEnabled: value })
    getApp().updateSettings({ soundEnabled: value })
  },

  toggleVibrate(e) {
    const value = e.detail.value
    this.setData({ vibrateEnabled: value })
    getApp().updateSettings({ vibrateEnabled: value })
  },

  showAbout() {
    wx.showModal({
      title: '关于财富自由棋',
      content: '版本：v1.0.0\n\n财富自由棋是一款财商教育类桌游，基于《穷爸爸富爸爸》现金流游戏理念设计。\n\n通过掷骰子走格子、购买资产增加被动收入，体验财务自由之路。',
      showCancel: false
    })
  },

  clearCache() {
    wx.showModal({
      title: '清除缓存',
      content: '确定要清除所有游戏缓存吗？这将删除所有存档和设置。',
      success: (res) => {
        if (res.confirm) {
          wx.clearStorageSync()
          this.setData({
            stats: { totalGames: 0, winGames: 0, maxFreedom: 0, totalAssets: 0 },
            achievements: achievements.map(ach => ({ ...ach, unlocked: false }))
          })
          wx.showToast({ title: '缓存已清除', icon: 'success' })
        }
      }
    })
  },

  logout() {
    wx.showModal({
      title: '退出登录',
      content: '确定要退出登录吗？',
      success: (res) => {
        if (res.confirm) {
          wx.removeStorageSync('userInfo')
          wx.removeStorageSync('selectedCareer')
          wx.removeStorageSync('gameSave')
          wx.reLaunch({ url: '/pages/home/home' })
        }
      }
    })
  }
})

const dailyRewards = [
  { day: 1, reward: 500, name: '初来乍到' },
  { day: 2, reward: 800, name: '再接再厉' },
  { day: 3, reward: 1000, name: '小有成就' },
  { day: 4, reward: 1500, name: '渐入佳境' },
  { day: 5, reward: 2000, name: '坚持不懈' },
  { day: 6, reward: 3000, name: '持之以恒' },
  { day: 7, reward: 5000, name: '签到大师' }
]

function isTodaySigned() {
  const lastSignDate = wx.getStorageSync('lastSignDate')
  const today = new Date().toDateString()
  return lastSignDate === today
}

function getSignInfo() {
  const streak = wx.getStorageSync('signStreak') || 0
  const todaySigned = isTodaySigned()
  
  let currentStreak = streak
  if (!todaySigned) {
    const lastSignDate = wx.getStorageSync('lastSignDate')
    if (lastSignDate) {
      const yesterday = new Date()
      yesterday.setDate(yesterday.getDate() - 1)
      if (lastSignDate !== yesterday.toDateString()) {
        currentStreak = 0
      }
    }
  }
  
  const nextReward = dailyRewards[(currentStreak % 7)]
  
  return {
    todaySigned,
    streak: currentStreak,
    nextReward: nextReward,
    allRewards: dailyRewards
  }
}

function doSign() {
  if (isTodaySigned()) {
    return { success: false, message: '今日已签到' }
  }
  
  let streak = wx.getStorageSync('signStreak') || 0
  const lastSignDate = wx.getStorageSync('lastSignDate')
  
  if (lastSignDate) {
    const yesterday = new Date()
    yesterday.setDate(yesterday.getDate() - 1)
    if (lastSignDate !== yesterday.toDateString()) {
      streak = 0
    }
  }
  
  streak++
  const reward = dailyRewards[(streak - 1) % 7]
  
  wx.setStorageSync('lastSignDate', new Date().toDateString())
  wx.setStorageSync('signStreak', streak)
  
  return { 
    success: true, 
    reward: reward,
    streak: streak,
    message: `签到第${streak}天，获得¥${reward.reward}！`
  }
}

module.exports = { dailyRewards, isTodaySigned, getSignInfo, doSign }
